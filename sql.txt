CREATE TABLE MovieCatalog (
    Movie_ID INT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1),
    Movie_Name VARCHAR(255),
    Rate VARCHAR(5),
    Year_No VARCHAR(4),
    Price DECIMAL
);

CREATE TABLE ShoppingCart (
    Item_ID INT NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1),
    Movie_ID INT NOT NULL,
    Quantity INT,
    FOREIGN KEY (Movie_ID) REFERENCES MOVIECATALOG(Movie_ID)
);
